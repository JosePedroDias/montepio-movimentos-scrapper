(function() {
    var trEls = document.querySelectorAll('#printable > table > tbody > tr:nth-child(6) tr');
    trEls = Array.prototype.slice.call(trEls);
    trEls.shift();
    trEls.pop();
    trEls.pop();
    trEls.pop();
    return trEls.map(function(trEl) {
        var tdEls = trEl.querySelectorAll('td');
        var aEl = tdEls[2].querySelector('a');
        var a = aEl.getAttribute('onClick');
        var b = a.split(/[\(\)]/)[1];
        b = b.substring(1, b.length-1);
        b = b.split("','");
        return {
            dataMov: tdEls[0].innerText,
            dataValor: tdEls[1].innerText,
            descricao: aEl.innerText,
            extra: b,
            importancia: tdEls[3].innerText,
            saldoCont: tdEls[4].innerText
        };
    });
})();
